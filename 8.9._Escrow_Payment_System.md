# <a name="_1w6k36gvztqh"></a> **Escrow Payment System** 
Design a smart contract that acts as an escrow between a buyer and a seller. Funds should be released only when both parties agree or after a certain time if no disputes are raised. -----

**STEP 1: CREATE FILE**

Create new file: Escrow.sol-----
# <a name="_4s3ist18e4t7"></a>**STEP 2: PASTE COMPLETE CODE**
// SPDX-License-Identifier: MIT

pragma solidity ^0.8.20;

contract Escrow {

`    `address public buyer;

`    `address payable public seller;

`    `uint256 public amount;

`    `uint256 public releaseTime;

`    `bool public buyerApproved;

`    `bool public sellerApproved;

`    `bool public fundsReleased;

`    `// 🏗 Constructor

`    `constructor(

`        `address payable \_seller,

`        `uint256 \_durationInMinutes

`    `) payable {

`        `buyer = msg.sender;

`        `seller = \_seller;

`        `amount = msg.value;

`        `releaseTime =

`            `block.timestamp + (\_durationInMinutes \* 1 minutes);

`    `}

`    `// ✅ Buyer Approval

`    `function approveByBuyer() public {

`        `require(msg.sender == buyer, "Only buyer allowed");

`        `buyerApproved = true;

`    `}

`    `// ✅ Seller Approval

`    `function approveBySeller() public {

`        `require(msg.sender == seller, "Only seller allowed");

`        `sellerApproved = true;

`    `}

`    `// 💰 Release Funds

`    `function releaseFunds() public {

`        `require(!fundsReleased, "Already released");

`        `require(

`            `(buyerApproved && sellerApproved)

`            `|| block.timestamp >= releaseTime,

`            `"Conditions not met"

`        `);

`        `fundsReleased = true;

`        `(bool sent, ) = seller.call{value: amount}("");

`        `require(sent, "Transfer failed");

`    `}

`    `// 🔄 Refund Buyer

`    `function refundBuyer() public {

`        `require(!fundsReleased, "Already released");

`        `require(

`            `msg.sender == buyer,

`            `"Only buyer can refund"

`        `);

`        `fundsReleased = true;

`        `(bool sent, ) =

`            `payable(buyer).call{value: amount}("");

`        `require(sent, "Refund failed");

`    `}

`    `// 📊 Contract Balance

`    `function getContractBalance()

`        `public

`        `view

`        `returns(uint256)

`    `{

`        `return address(this).balance;

`    `}

}
