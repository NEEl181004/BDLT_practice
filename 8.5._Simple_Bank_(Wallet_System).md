# <a name="_47h4zlwy1ikj"></a>**Simple Bank (Wallet System)** 

Build a smart contract that functions as a simple bank. Users should be able to deposit, withdraw, and check their balance. Only the account holder should have access to their funds. 

-----
# <a name="_pgf5leuj4qxs"></a>**STEP 1: CREATE FILE**
Open: Remix IDE

Create new file: SimpleBank.sol-----
# <a name="_xq0i865g6iof"></a>**STEP 2: PASTE COMPLETE SMART CONTRACT CODE**
// SPDX-License-Identifier: MIT

pragma solidity ^0.8.20;

contract SimpleBank {

`    `// Stores balance of each user

`    `mapping(address => uint256) private balances;

`    `// 💰 Deposit ETH

`    `function deposit() public payable {

`        `require(msg.value > 0, "Send some ETH");

`        `balances[msg.sender] += msg.value;

`    `}

`    `// 💸 Withdraw ETH

`    `function withdraw(uint256 \_amount) public {

`        `require(

`            `balances[msg.sender] >= \_amount,

`            `"Insufficient balance"

`        `);

`        `balances[msg.sender] -= \_amount;

`        `(bool sent, ) = payable(msg.sender).call{value: \_amount}("");

`        `require(sent, "Withdrawal failed");

`    `}

`    `// 📊 Check Balance

`    `function checkBalance()

`        `public

`        `view

`        `returns (uint256)

`    `{

`        `return balances[msg.sender];

`    `}

}
