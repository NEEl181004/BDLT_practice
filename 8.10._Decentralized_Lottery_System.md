// SPDX-License-Identifier: MIT

pragma solidity ^0.8.20;

contract Lottery {

`    `address public manager;

`    `address[] public players;

`    `constructor() {

`        `manager = msg.sender;

`    `}

`    `// 🎟 Buy Ticket

`    `function buyTicket() public payable {

`        `require(

`            `msg.value == 1 ether,

`            `"Ticket price is 1 ETH"

`        `);

`        `players.push(msg.sender);

`    `}

`    `// 📊 Get Players

`    `function getPlayers()

`        `public

`        `view

`        `returns(address[] memory)

`    `{

`        `return players;

`    `}

`    `// 💰 Contract Balance

`    `function getBalance()

`        `public

`        `view

`        `returns(uint256)

`    `{

`        `return address(this).balance;

`    `}

`    `// 🎲 Generate Random Number

`    `function random()

`        `private

`        `view

`        `returns(uint256)

`    `{

`        `return uint256(

`            `keccak256(

`                `abi.encodePacked(

`                    `block.timestamp,

`                    `block.prevrandao,

`                    `players.length

`                `)

`            `)

`        `);

`    `}

`    `// 🏆 Pick Winner

`    `function pickWinner() public {

`        `require(

`            `msg.sender == manager,

`            `"Only manager can pick winner"

`        `);

`        `require(

`            `players.length > 0,

`            `"No players joined"

`        `);

`        `uint256 winnerIndex =

`            `random() % players.length;

`        `address payable winner =

`            `payable(players[winnerIndex]);

`        `// ✅ Updated payment method

`        `(bool sent, ) =

`            `winner.call{value: address(this).balance}("");

`        `require(sent, "Transfer failed");

`        `// Reset lottery

`        `players = new address[](0);

`    `}

}
