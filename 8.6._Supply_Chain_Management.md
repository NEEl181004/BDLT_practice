# <a name="_byjis5g1jgz1"></a> **STEP 2: PASTE COMPLETE SMART CONTRACT CODE**
// SPDX-License-Identifier: MIT

pragma solidity ^0.8.20;

contract SupplyChain {

`    `// 📍 Structure for tracking steps

`    `struct Step {

`        `string location;

`        `uint256 timestamp;

`        `address participant;

`    `}

`    `// 📦 Product structure

`    `struct Product {

`        `uint256 productId;

`        `string productName;

`        `Step[] steps;

`    `}

`    `// productId => Product

`    `mapping(uint256 => Product) private products;

`    `// Check if product exists

`    `mapping(uint256 => bool) public productExists;

`    `// ➕ Add Product

`    `function addProduct(

`        `uint256 \_productId,

`        `string memory \_productName

`    `) public {

`        `require(

`            `!productExists[\_productId],

`            `"Product already exists"

`        `);

`        `Product storage p = products[\_productId];

`        `p.productId = \_productId;

`        `p.productName = \_productName;

`        `productExists[\_productId] = true;

`    `}

`    `// 🚚 Add Supply Chain Step

`    `function addStep(

`        `uint256 \_productId,

`        `string memory \_location

`    `) public {

`        `require(

`            `productExists[\_productId],

`            `"Product does not exist"

`        `);

`        `products[\_productId].steps.push(

`            `Step(

`                `\_location,

`                `block.timestamp,

`                `msg.sender

`            `)

`        `);

`    `}

`    `// 📊 Get Product Basic Details

`    `function getProduct(uint256 \_productId)

`        `public

`        `view

`        `returns (

`            `uint256,

`            `string memory,

`            `uint256

`        `)

`    `{

`        `require(

`            `productExists[\_productId],

`            `"Product does not exist"

`        `);

`        `Product storage p = products[\_productId];

`        `return (

`            `p.productId,

`            `p.productName,

`            `p.steps.length

`        `);

`    `}

`    `// 📍 Get Specific Step Details

`    `function getStep(

`        `uint256 \_productId,

`        `uint256 \_stepIndex

`    `)

`        `public

`        `view

`        `returns (

`            `string memory,

`            `uint256,

`            `address

`        `)

`    `{

`        `require(

`            `productExists[\_productId],

`            `"Product does not exist"

`        `);

`        `Step storage s = products[\_productId].steps[\_stepIndex];

`        `return (

`            `s.location,

`            `s.timestamp,

`            `s.participant

`        `);

`    `}

}
