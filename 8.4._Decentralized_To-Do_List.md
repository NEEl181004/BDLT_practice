# <a name="_x46efp74t55w"></a>Develop a smart contract that allows users to add, update, and mark tasks as completed. Each user's tasks should be stored independently and be accessible only by the user.

` `**STEP 2: PASTE COMPLETE SMART CONTRACT CODE**

// SPDX-License-Identifier: MIT

pragma solidity ^0.8.20;

contract TodoList {

`    `struct Task {

`        `uint256 id;

`        `string taskText;

`        `bool completed;

`    `}

`    `// Each user has their own tasks

`    `mapping(address => Task[]) private userTasks;

`    `// ➕ Add Task

`    `function addTask(string memory \_taskText) public {

`        `uint256 taskId = userTasks[msg.sender].length;

`        `userTasks[msg.sender].push(

`            `Task(taskId, \_taskText, false)

`        `);

`    `}

`    `// ✏️ Update Task

`    `function updateTask(

`        `uint256 \_taskId,

`        `string memory \_newText

`    `) public {

`        `require(

`            `\_taskId < userTasks[msg.sender].length,

`            `"Task does not exist"

`        `);

`        `userTasks[msg.sender][\_taskId].taskText = \_newText;

`    `}

`    `// ✅ Mark Task Completed

`    `function markCompleted(uint256 \_taskId) public {

`        `require(

`            `\_taskId < userTasks[msg.sender].length,

`            `"Task does not exist"

`        `);

`        `userTasks[msg.sender][\_taskId].completed = true;

`    `}

`    `// 📊 Get Task

`    `function getTask(uint256 \_taskId)

`        `public

`        `view

`        `returns (

`            `uint256 id,

`            `string memory taskText,

`            `bool completed

`        `)

`    `{

`        `require(

`            `\_taskId < userTasks[msg.sender].length,

`            `"Task does not exist"

`        `);

`        `Task memory t = userTasks[msg.sender][\_taskId];

`        `return (

`            `t.id,

`            `t.taskText,

`            `t.completed

`        `);

`    `}

`    `// 📋 Get Total Tasks

`    `function getTaskCount()

`        `public

`        `view

`        `returns (uint256)

`    `{

`        `return userTasks[msg.sender].length;

`    `}

}
