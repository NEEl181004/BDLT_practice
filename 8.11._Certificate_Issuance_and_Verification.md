# <a name="_uman51fjugnr"></a>**STEP 2: PASTE COMPLETE SMART CONTRACT CODE**
// SPDX-License-Identifier: MIT

pragma solidity ^0.8.20;

contract CertificateVerification {

`    `// 👨‍💼 Admin Address

`    `address public admin;

`    `constructor() {

`        `admin = msg.sender;

`    `}

`    `// 🎓 Certificate Structure

`    `struct Certificate {

`        `uint256 certificateId;

`        `string studentName;

`        `string courseName;

`        `uint256 issueYear;

`        `address studentAddress;

`        `bool exists;

`    `}

`    `// certificateId => Certificate

`    `mapping(uint256 => Certificate) public certificates;

`    `// Prevent duplicate IDs

`    `mapping(uint256 => bool) public certificateIssued;

`    `// 🎓 Issue Certificate

`    `function issueCertificate(

`        `uint256 \_certificateId,

`        `string memory \_studentName,

`        `string memory \_courseName,

`        `uint256 \_issueYear,

`        `address \_studentAddress

`    `) public {

`        `require(

`            `msg.sender == admin,

`            `"Only admin can issue certificates"

`        `);

`        `require(

`            `!certificateIssued[\_certificateId],

`            `"Certificate already exists"

`        `);

`        `certificates[\_certificateId] = Certificate(

`            `\_certificateId,

`            `\_studentName,

`            `\_courseName,

`            `\_issueYear,

`            `\_studentAddress,

`            `true

`        `);

`        `certificateIssued[\_certificateId] = true;

`    `}

`    `// ✅ Verify Certificate

`    `function verifyCertificate(uint256 \_certificateId)

`        `public

`        `view

`        `returns (

`            `uint256,

`            `string memory,

`            `string memory,

`            `uint256,

`            `address,

`            `bool

`        `)

`    `{

`        `require(

`            `certificateIssued[\_certificateId],

`            `"Certificate does not exist"

`        `);

`        `Certificate memory cert =

`            `certificates[\_certificateId];

`        `return (

`            `cert.certificateId,

`            `cert.studentName,

`            `cert.courseName,

`            `cert.issueYear,

`            `cert.studentAddress,

`            `cert.exists

`        `);

`    `}

}
