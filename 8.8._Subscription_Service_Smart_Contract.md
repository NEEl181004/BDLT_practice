
# <a name="_d24rykbt6ezw"></a>**Subscription Service Smart Contract**

Implement a smart contract that manages subscription-based access to a service. Users should pay to subscribe for a period of time, and access should automatically expire afterward.

# <a name="_oswczvxpfzwp"></a>**STEP 2: PASTE COMPLETE SMART CONTRACT CODE**
// SPDX-License-Identifier: MIT

pragma solidity ^0.8.20;

contract SubscriptionService {

`    `address public owner;

`    `uint256 public subscriptionFee = 1 ether;

`    `uint256 public subscriptionDuration = 1 minutes;

`    `// user => expiry timestamp

`    `mapping(address => uint256) public subscriptionExpiry;

`    `constructor() {

`        `owner = msg.sender;

`    `}

`    `// 💳 Subscribe

`    `function subscribe() public payable {

`        `require(

`            `msg.value == subscriptionFee,

`            `"Incorrect subscription fee"

`        `);

`        `// Extend subscription

`        `if (subscriptionExpiry[msg.sender] < block.timestamp) {

`            `subscriptionExpiry[msg.sender] =

`                `block.timestamp + subscriptionDuration;

`        `} else {

`            `subscriptionExpiry[msg.sender] += subscriptionDuration;

`        `}

`    `}

`    `// ✅ Check Subscription Status

`    `function isSubscribed()

`        `public

`        `view

`        `returns (bool)

`    `{

`        `return subscriptionExpiry[msg.sender] > block.timestamp;

`    `}

`    `// 📅 Get Expiry Time

`    `function getExpiryTime()

`        `public

`        `view

`        `returns (uint256)

`    `{

`        `return subscriptionExpiry[msg.sender];

`    `}

`    `// 💰 Withdraw Contract Funds

`    `function withdrawFunds() public {

`        `require(msg.sender == owner, "Only owner can withdraw");

`        `uint256 amount = address(this).balance;

`        `(bool sent, ) = payable(owner).call{value: amount}("");

`        `require(sent, "Withdrawal failed");

`    `}

}
