
# <a name="_8sb92usdfadh"></a>**Academic Certificate Verification System** 
Design and deploy smart contract for Academic Certificate Verification System: 

A The university wants to issue tamper-proof digital certificates to graduates. Employers should be able to verify authenticity through a public blockchain without contacting the university. 

Smart Contract Requirements: Only university admin can issue certificates.Store student ID, name, course, year, and certificate hash.Public function for verification of certificate authenticity.Prevent duplicate certificate entries. Handle both batch issuance and single issuance.


**SMART CONTRACT CODE**

Paste this directly into Remix IDE

// SPDX-License-Identifier: MIT

pragma solidity ^0.8.20;

contract CertificateVerification {

`    `address public admin;

`    `constructor() {

`        `admin = msg.sender;

`    `}

`    `struct Certificate {

`        `string studentId;

`        `string name;

`        `string course;

`        `uint256 year;

`        `string certHash;

`        `bool exists;

`    `}

`    `mapping(string => Certificate) private certificates;

`    `// 🔐 Only admin can issue

`    `modifier onlyAdmin() {

`        `require(msg.sender == admin, "Not authorized");

`        `\_;

`    `}

`    `// 📌 Issue single certificate

`    `function issueCertificate(

`        `string memory studentId,

`        `string memory name,

`        `string memory course,

`        `uint256 year,

`        `string memory certHash

`    `) public onlyAdmin {

`        `require(!certificates[certHash].exists, "Duplicate certificate");

`        `certificates[certHash] = Certificate(

`            `studentId,

`            `name,

`            `course,

`            `year,

`            `certHash,

`            `true

`        `);

`    `}

`    `// 📦 Batch issuance (FIXED VARIABLE NAME)

`    `function issueBatch(

`        `string[] memory studentIds,

`        `string[] memory names,

`        `string[] memory courses,

`        `uint256[] memory yearList,

`        `string[] memory certHashes

`    `) public onlyAdmin {

`        `require(

`            `studentIds.length == names.length &&

`            `names.length == courses.length &&

`            `courses.length == yearList.length &&

`            `yearList.length == certHashes.length,

`            `"Array length mismatch"

`        `);

`        `for (uint i = 0; i < studentIds.length; i++) {

`            `require(!certificates[certHashes[i]].exists, "Duplicate certificate");

`            `certificates[certHashes[i]] = Certificate(

`                `studentIds[i],

`                `names[i],

`                `courses[i],

`                `yearList[i],

`                `certHashes[i],

`                `true

`            `);

`        `}

`    `}

`    `// 🔍 Verify certificate (public)

`    `function verifyCertificate(string memory certHash)

`        `public

`        `view

`        `returns (

`            `string memory studentId,

`            `string memory name,

`            `string memory course,

`            `uint256 year,

`            `bool valid

`        `)

`    `{

`        `require(certificates[certHash].exists, "Certificate not found");

`        `Certificate memory c = certificates[certHash];

`        `return (

`            `c.studentId,

`            `c.name,

`            `c.course,

`            `c.year,

`            `c.exists

`        `);

`    `}

}
