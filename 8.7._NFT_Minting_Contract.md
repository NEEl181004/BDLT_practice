// SPDX-License-Identifier: MIT

pragma solidity ^0.8.20;

contract NFTMinting {

`    `struct NFT {

`        `uint256 tokenId;

`        `string name;

`        `string metadataURI;

`        `address owner;

`    `}

`    `uint256 public nextTokenId;

`    `// tokenId => NFT

`    `mapping(uint256 => NFT) public nfts;

`    `// owner => token count

`    `mapping(address => uint256) public balances;

`    `// 🎨 Mint NFT

`    `function mintNFT(

`        `string memory \_name,

`        `string memory \_metadataURI

`    `) public {

`        `nextTokenId++;

`        `nfts[nextTokenId] = NFT(

`            `nextTokenId,

`            `\_name,

`            `\_metadataURI,

`            `msg.sender

`        `);

`        `balances[msg.sender]++;

`    `}

`    `// 🔄 Transfer NFT

`    `function transferNFT(

`        `address \_to,

`        `uint256 \_tokenId

`    `) public {

`        `require(

`            `nfts[\_tokenId].owner == msg.sender,

`            `"You are not owner"

`        `);

`        `address previousOwner = nfts[\_tokenId].owner;

`        `nfts[\_tokenId].owner = \_to;

`        `balances[previousOwner]--;

`        `balances[\_to]++;

`    `}

`    `// 📊 Get NFT Details

`    `function getNFT(uint256 \_tokenId)

`        `public

`        `view

`        `returns (

`            `uint256,

`            `string memory,

`            `string memory,

`            `address

`        `)

`    `{

`        `NFT memory nft = nfts[\_tokenId];

`        `return (

`            `nft.tokenId,

`            `nft.name,

`            `nft.metadataURI,

`            `nft.owner

`        `);

`    `}

`    `// 👤 Check NFT Balance

`    `function checkBalance()

`        `public

`        `view

`        `returns (uint256)

`    `{

`        `return balances[msg.sender];

`    `}

}
