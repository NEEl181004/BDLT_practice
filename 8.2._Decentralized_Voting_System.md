# <a name="_1otx4zbg2b0g"></a> **Decentralized Voting System** 

Design and implement a smart contract in Remix that allows users to vote for candidates in a secure and transparent election. The system should ensure one vote per address and allow the admin to start and end the voting period.

// SPDX-License-Identifier: MIT

pragma solidity ^0.8.20;

contract DecentralizedVoting {

`    `address public admin;

`    `bool public votingActive;

`    `constructor() {

`        `admin = msg.sender;

`        `votingActive = false;

`    `}

`    `struct Candidate {

`        `string name;

`        `uint256 voteCount;

`    `}

`    `Candidate[] public candidates;

`    `mapping(address => bool) public hasVoted;

`    `modifier onlyAdmin() {

`        `require(msg.sender == admin, "Not admin");

`        `\_;

`    `}

`    `modifier onlyDuringVoting() {

`        `require(votingActive == true, "Voting not active");

`        `\_;

`    `}

`    `// Add single candidate

`    `function addCandidate(string memory name) public onlyAdmin {

`        `candidates.push(Candidate(name, 0));

`    `}

`    `// Add multiple candidates

`    `function addMultipleCandidates(string[] memory names) public onlyAdmin {

`        `for (uint i = 0; i < names.length; i++) {

`            `candidates.push(Candidate(names[i], 0));

`        `}

`    `}

`    `// Start voting

`    `function startVoting() public onlyAdmin {

`        `votingActive = true;

`    `}

`    `// End voting

`    `function endVoting() public onlyAdmin {

`        `votingActive = false;

`    `}

`    `// Vote

`    `function vote(uint256 candidateIndex) public onlyDuringVoting {

`        `require(!hasVoted[msg.sender], "Already voted");

`        `require(candidateIndex < candidates.length, "Invalid candidate");

`        `hasVoted[msg.sender] = true;

`        `candidates[candidateIndex].voteCount++;

`    `}

`    `// Get candidate result

`    `function getCandidate(uint256 index)

`        `public

`        `view

`        `returns (string memory name, uint256 votes)

`    `{

`        `Candidate memory c = candidates[index];

`        `return (c.name, c.voteCount);

`    `}

`    `// Get total candidates

`    `function getCandidatesCount() public view returns (uint256) {

`        `return candidates.length;

`    `}

}
